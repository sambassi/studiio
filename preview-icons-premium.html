Structure React — Composant PlatformIcon
Le composant PlatformIcon remplace les emojis dans PLATFORM_SAFE_ZONES et dans le sélecteur de plateformes. Chaque plateforme a sa propre couleur de glow et son SVG path officiel.
// src/components/ui/PlatformIcon.tsx const PLATFORM_CONFIGS = { instagram: { color: '#E1306C', label: 'Instagram', path: 'M12 2.163c3.204 0 3.584...', // SVG path }, tiktok: { color: '#00F2EA', label: 'TikTok', path: 'M19.59 6.69a4.83...', }, youtube: { color: '#FF0000', label: 'YouTube', path: 'M23.498 6.186...', }, facebook: { color: '#1877F2', label: 'Facebook', path: 'M24 12.073c0-6.627...', }, }; interface PlatformIconProps { platform: 'instagram' | 'tiktok' | 'youtube' | 'facebook'; isActive?: boolean; size?: 'sm' | 'md'; // sm = mini-icon, md = capsule showLabel?: boolean; onClick?: () => void; } export function PlatformIcon({ platform, isActive, size = 'md', showLabel = true, onClick }: PlatformIconProps) { const config = PLATFORM_CONFIGS[platform]; if (size === 'sm') { return ( <button className={`mini-icon ${platform} ${isActive ? 'active' : ''}`} onClick={onClick} > <svg viewBox="0 0 24 24" fill={isActive ? config.color : '#888'}> <path d={config.path} /> </svg> </button> ); } return ( <button className={`icon-capsule ${platform} ${isActive ? 'active' : ''}`} onClick={onClick} > <div className={`icon-circle ${platform}`}> <svg viewBox="0 0 24 24"> <path d={config.path} fill={config.color} /> </svg> </div> {showLabel && <span className="icon-label">{config.label}</span>} </button> ); }
// Mise à jour de PLATFORM_SAFE_ZONES // Avant: const PLATFORM_SAFE_ZONES = { instagram: { icon: '📸', ... }, tiktok: { icon: '🎵', ... }, }; // Après: le champ `icon` est remplacé par `platform` (clé du composant) const PLATFORM_SAFE_ZONES = { instagram: { platform: 'instagram', color: '#E1306C', ... }, tiktok: { platform: 'tiktok', color: '#00F2EA', ... }, youtube: { platform: 'youtube', color: '#FF0000', ... }, facebook: { platform: 'facebook', color: '#1877F2', ... }, }; // Usage dans le sélecteur: {Object.entries(PLATFORM_SAFE_ZONES).map(([key, zone]) => ( <PlatformIcon key={key} platform={zone.platform} isActive={safeZonePlatform === key} size="sm" onClick={() => setSafeZonePlatform(key)} /> ))}
